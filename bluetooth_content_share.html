  { name:'Draw',    type:'pct', value:50, color:'#c8f55a' },

// ── State ─────────────────────────────────────────────────────────
let buckets  = JSON.parse(localStorage.getItem('buckets_v2') || 'null') || DEFAULT_BUCKETS;
let currency = localStorage.getItem('currency') || '$';
let theme    = localStorage.getItem('theme')    || 'dark';

applyTheme(theme, false);
applyCurrency(currency, false);

// ── Helpers ───────────────────────────────────────────────────────
function fmt(n) {
  if (n === 0) return currency + '0';
  return currency + Math.abs(n).toLocaleString('en-US', { minimumFractionDigits:2, maximumFractionDigits:2 });
}

function persist() {
  localStorage.setItem('buckets_v2', JSON.stringify(buckets));
  localStorage.setItem('currency', currency);
  localStorage.setItem('theme', theme);
}

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2200);
}

// ── Calculation logic ─────────────────────────────────────────────
function calcSplit(amount) {
  const fixedBuckets = buckets.filter(b => b.type === 'fixed');
  const pctBuckets   = buckets.filter(b => b.type === 'pct');

  const totalFixed = fixedBuckets.reduce((s, b) => s + (parseFloat(b.value) || 0), 0);
  const remainder  = amount - totalFixed;
  const isOverdraft = totalFixed > amount && amount > 0;

  // Each bucket gets its calculated amount
  const results = buckets.map(b => {
    if (b.type === 'fixed') {
      return parseFloat(b.value) || 0;
    } else {
      // % of remainder (could be negative if overdraft)
      return remainder * ((parseFloat(b.value) || 0) / 100);
    }
  });

  return { results, totalFixed, remainder, isOverdraft };
}

// ── Theme ─────────────────────────────────────────────────────────
function applyTheme(t, doSave = true) {
  theme = t;
  document.documentElement.setAttribute('data-theme', t);
  document.getElementById('themeToggle').textContent = t === 'dark' ? '🌙' : '☀️';
  document.getElementById('themeColor').content = t === 'dark' ? '#0f1117' : '#f4f5f7';
  document.querySelectorAll('.theme-btn').forEach(b => b.classList.toggle('selected', b.dataset.theme === t));
  if (doSave) persist();
}

document.getElementById('themeToggle').addEventListener('click', () => applyTheme(theme === 'dark' ? 'light' : 'dark'));
document.querySelectorAll('.theme-btn').forEach(b => b.addEventListener('click', () => applyTheme(b.dataset.theme)));

// ── Currency ──────────────────────────────────────────────────────
function applyCurrency(sym, doSave = true) {
  currency = sym;
  document.getElementById('currencyDisplay').textContent = sym;
  document.querySelectorAll('.currency-btn').forEach(b => b.classList.toggle('selected', b.dataset.sym === sym));
  updateCalc();
  if (doSave) persist();
}

function buildCurrencyGrid() {
  const grid = document.getElementById('currencyGrid');
  grid.innerHTML = '';
  CURRENCIES.forEach(sym => {
    const btn = document.createElement('button');
    btn.className = 'currency-btn' + (sym === currency ? ' selected' : '');
    btn.dataset.sym = sym;
    btn.innerHTML = `<span class="sym">${sym}</span>`;
    btn.addEventListener('click', () => applyCurrency(sym));
    grid.appendChild(btn);
  });
}

  });
});

// ── CALCULATE VIEW ────────────────────────────────────────────────
function renderCalc() {
  const display  = document.getElementById('bucketsDisplay');
  const barTrack = document.getElementById('barTrack');
  document.querySelectorAll('style.dyn').forEach(s => s.remove());
  display.innerHTML = '';
  barTrack.innerHTML = '';

  const amount = parseFloat(document.getElementById('amountInput').value) || 0;
  const { results, totalFixed, remainder, isOverdraft } = calcSplit(amount);

  // Section labels
  const hasFixed = buckets.some(b => b.type === 'fixed');
  const hasPct   = buckets.some(b => b.type === 'pct');

  buckets.forEach((b, i) => {
    // Optional section divider label
    if (hasFixed && hasPct) {
      if (i === 0 && b.type === 'fixed') {
        const lbl = document.createElement('p');
        lbl.className = 'bucket-section-label';
        lbl.textContent = 'Fixed amounts';
        display.appendChild(lbl);
      }
      if (b.type === 'pct' && (i === 0 || buckets[i-1].type === 'fixed')) {
        const lbl = document.createElement('p');
        lbl.className = 'bucket-section-label';
        lbl.textContent = 'Percent of remainder';
        display.appendChild(lbl);
      }
    }

    const val = results[i];
    const el  = document.createElement('div');
    el.className = 'bucket' + (b.type === 'fixed' ? ' type-fixed' : '');
    el.id = 'bucket-' + i;

    const subLabel = b.type === 'fixed'
      ? `fixed ${fmt(parseFloat(b.value)||0)}`
      : `${b.value}% of remainder`;

    el.innerHTML = `
      <div class="bucket-left">
        <span class="bucket-name" style="color:${b.color}">${b.name}</span>
        <span class="bucket-sub">${subLabel}</span>
      </div>
      <span class="bucket-amount ${amount === 0 ? 'zero' : ''}" id="amt-${i}">${fmt(val)}</span>
    `;

    const s = document.createElement('style');
    s.className = 'dyn';
    s.textContent = `#bucket-${i}::after { background: ${b.color}; }`;
    document.head.appendChild(s);
    display.appendChild(el);

    // Bar segments — width based on % of total invoice
    const seg = document.createElement('div');
    seg.className = 'bar-seg';
    seg.id = 'bar-' + i;
    seg.style.background = b.color;
    const barPct = amount > 0 ? Math.max(0, (val / amount) * 100) : (b.type === 'pct' ? b.value : 0);
    seg.style.width   = barPct + '%';
    seg.style.opacity = amount > 0 ? '1' : '0.25';
    barTrack.appendChild(seg);
  });

  updateSummary(amount, totalFixed, remainder, isOverdraft, results);
}

function updateCalc() {
  const amount = parseFloat(document.getElementById('amountInput').value) || 0;
  const { results, totalFixed, remainder, isOverdraft } = calcSplit(amount);

  buckets.forEach((b, i) => {
    const val   = results[i];
    const amtEl = document.getElementById('amt-' + i);
    const barEl = document.getElementById('bar-' + i);
    if (!amtEl) return;

    amtEl.textContent = fmt(val);
    amtEl.classList.toggle('zero', amount === 0);

    if (amount > 0) {
      amtEl.classList.remove('pop'); void amtEl.offsetWidth; amtEl.classList.add('pop');
      setTimeout(() => amtEl.classList.remove('pop'), 400);
      const bEl = document.getElementById('bucket-' + i);
      if (bEl) { bEl.classList.remove('updated'); void bEl.offsetWidth; bEl.classList.add('updated'); setTimeout(() => bEl.classList.remove('updated'), 500); }
    }

    if (barEl) {
      const barPct = amount > 0 ? Math.max(0, (val / amount) * 100) : (b.type === 'pct' ? b.value : 0);
      barEl.style.width   = barPct + '%';
      barEl.style.opacity = amount > 0 ? '1' : '0.25';
    }
  });

  updateSummary(amount, totalFixed, remainder, isOverdraft, results);
}

function updateSummary(amount, totalFixed, remainder, isOverdraft, results) {
  const hasFixed   = buckets.some(b => b.type === 'fixed');
  const totalAlloc = results.reduce((s, v) => s + v, 0);

  const sfEl = document.getElementById('sumFixed');
  const srEl = document.getElementById('sumRemaining');
  const stEl = document.getElementById('sumTotal');
  const wb   = document.getElementById('warningBanner');
  const wt   = document.getElementById('warningText');

  if (amount > 0) {
    if (hasFixed) {
      sfEl.textContent = fmt(totalFixed);
      sfEl.classList.add('visible');
      sfEl.classList.toggle('warn', isOverdraft);

      srEl.textContent = isOverdraft ? '-' + fmt(Math.abs(remainder)) : fmt(Math.max(0, remainder));
      srEl.classList.add('visible');
      srEl.classList.toggle('warn', isOverdraft);
    } else {
      sfEl.classList.remove('visible');
      srEl.classList.remove('visible');
    }

    stEl.textContent = fmt(totalAlloc) + ' ✓';
    stEl.classList.add('visible');
    stEl.classList.remove('warn');

    // Warning banner
    if (isOverdraft) {
      wb.classList.add('show');
      wt.textContent = `⚠ Fixed deductions (${fmt(totalFixed)}) exceed this invoice by ${fmt(Math.abs(remainder))}. Percent buckets will show a shortfall.`;
    } else {
      wb.classList.remove('show');
    }
  } else {
    [sfEl, srEl, stEl].forEach(el => el.classList.remove('visible'));
    wb.classList.remove('show');
  }
}

document.getElementById('amountInput').addEventListener('input', updateCalc);
document.getElementById('clearBtn').addEventListener('click', () => {
  document.getElementById('amountInput').value = '';
  updateCalc();
  document.getElementById('amountInput').focus();
});

// ── SETTINGS VIEW ─────────────────────────────────────────────────
function renderSettings() {
  const rows = document.getElementById('bucketRows');
  rows.innerHTML = '';
  document.querySelectorAll('style.dyn-row').forEach(s => s.remove());

  buckets.forEach((b, i) => {
    const row = document.createElement('div');
    row.className = 'bucket-row';
    row.id = 'row-' + i;
    row.style.animationDelay = (i * 0.04) + 's';

    const s = document.createElement('style');
    s.className = 'dyn-row';
    s.textContent = `#row-${i}::before { background: ${b.color}; }`;
    document.head.appendChild(s);

    const isPct = b.type === 'pct';

    row.innerHTML = `
      <div class="bucket-row-top">
        <div class="color-dot" id="dot-${i}" style="background:${b.color}"></div>
        <input class="bucket-name-input" id="name-${i}" value="${b.name}" placeholder="Name" maxlength="14"/>
        <button class="delete-btn" data-index="${i}">×</button>
      </div>
      <div class="bucket-row-bottom">
        <div class="type-toggle">
          <button class="type-btn ${isPct ? 'active' : ''}" data-type="pct" data-index="${i}">%</button>
          <button class="type-btn ${!isPct ? 'active' : ''}" data-type="fixed" data-index="${i}">${currency}</button>
        </div>
        <div class="value-input-wrap">
          <span class="value-sym">${isPct ? '%' : currency}</span>
          <input class="value-input" id="val-${i}" type="number" min="0" value="${b.value}" inputmode="decimal"/>
        </div>
      </div>
    `;

    rows.appendChild(row);

    // Name
    row.querySelector('.bucket-name-input').addEventListener('input', e => { buckets[i].name = e.target.value; });

    // Value
    row.querySelector('.value-input').addEventListener('input', e => {
      buckets[i].value = parseFloat(e.target.value) || 0;
      updateTotalWarning();
    });

    // Type toggle
    row.querySelectorAll('.type-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        buckets[i].type  = btn.dataset.type;
        buckets[i].value = 0;
        renderSettings();
        updateTotalWarning();
      });
    });

    // Delete
    row.querySelector('.delete-btn').addEventListener('click', () => {
      if (buckets.length <= 1) { showToast('Need at least one bucket'); return; }
      buckets.splice(i, 1);
      renderSettings();
    });

    // Color dot
    row.querySelector('.color-dot').addEventListener('click', e => {
      e.stopPropagation();
      openColorPicker(i, row.querySelector('.color-dot'));
    });
  });

  updateTotalWarning();
}

function updateTotalWarning() {
  const pctBuckets = buckets.filter(b => b.type === 'pct');
  const total = pctBuckets.reduce((s, b) => s + (parseFloat(b.value) || 0), 0);
  const el = document.getElementById('totalWarning');

  if (pctBuckets.length === 0) {
    el.textContent = 'No % buckets';
    el.className = 'total-warning ok';
  } else if (Math.abs(total - 100) < 0.01) {
    el.textContent = '% = 100 ✓';
    el.className = 'total-warning ok';
  } else if (total > 100) {
    el.textContent = `% = ${total.toFixed(1)} — over`;
    el.className = 'total-warning over';
  } else {
    el.textContent = `% = ${total.toFixed(1)} — ${(100-total).toFixed(1)} left`;
    el.className = 'total-warning under';
  }
}

document.getElementById('addBucketBtn').addEventListener('click', () => {
  const used = buckets.map(b => b.color);
  const col  = PALETTE.find(c => !used.includes(c)) || PALETTE[buckets.length % PALETTE.length];
  buckets.push({ name:'Bucket', type:'pct', value:0, color:col });
  renderSettings();
  setTimeout(() => {
    const lastVal = document.getElementById('val-' + (buckets.length - 1));
    if (lastVal) { lastVal.focus(); lastVal.select(); }
  }, 80);
});

document.getElementById('saveBtn').addEventListener('click', () => {
  const pctBuckets = buckets.filter(b => b.type === 'pct');
  if (pctBuckets.length > 0) {
    const total = pctBuckets.reduce((s, b) => s + (parseFloat(b.value) || 0), 0);
    if (Math.abs(total - 100) > 0.5) {
      showToast(`% buckets must total 100 (now ${total.toFixed(1)})`);
      return;
    }
  }
  buckets = buckets.map(b => ({ ...b, value: parseFloat(b.value) || 0 }));
  persist();
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
  document.querySelector('[data-tab="calc"]').classList.add('active');
  document.getElementById('view-calc').classList.add('active');
  renderCalc();
  updateCalc();
  showToast('Saved ✓');
});

// ── Color Picker ──────────────────────────────────────────────────
function openColorPicker(index, dotEl) {
  const picker  = document.getElementById('colorPicker');
  const overlay = document.getElementById('overlay');
  picker.innerHTML = '';
  PALETTE.forEach(color => {
    const sw = document.createElement('div');
    sw.className = 'color-swatch' + (buckets[index].color === color ? ' selected' : '');
    sw.style.background = color;
    sw.addEventListener('click', () => { buckets[index].color = color; renderSettings(); closeColorPicker(); });
    picker.appendChild(sw);
  });
  const rect = dotEl.getBoundingClientRect();
  picker.style.top  = (rect.bottom + 8) + 'px';
  picker.style.left = Math.min(rect.left, window.innerWidth - 196) + 'px';
  picker.classList.remove('hidden');
  overlay.classList.add('active');
}

function closeColorPicker() {
  document.getElementById('colorPicker').classList.add('hidden');
  document.getElementById('overlay').classList.remove('active');
}

document.getElementById('overlay').addEventListener('click', closeColorPicker);
